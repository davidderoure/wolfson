%% --- INFO PANEL -------------------------------------------------------------
\section{Last phrase panel \textnormal{\small(the main information panel)}}

\begin{dashpanel}{last phrase \quad{[bass]}}
\textcolor{codecyan}{BUILDING}~~~\textcolor{codecyan}{modal}~~~~~~~~~scale~~\textcolor{stagebuilding}{BASS}~~~~~~~~{->}~~neutral\\
lead~~~bass~~~~\textcolor{codeyellow}{chord~~Dm}~~~~vel~80~~~n~8~~~~\textcolor{codecyan}{C~D~E~F~G~A~B}
\end{dashpanel}

\vspace{6pt}
\noindent\textbf{Panel title} --- what triggered this phrase:\\[2pt]
\hspace*{1em}\field{[bass]} = sax responded to a bass phrase you played.\\
\hspace*{1em}\field{[proactive]} = sax initiated during a silence.

\vspace{6pt}
\noindent\textbf{Top row:}

\vspace{2pt}
\begin{description}
  \item[stage name (large, coloured)]
    Current arc stage --- colour matches the arc bar above.
  \item[harmonic mode]
    The sax's harmonic strategy: \val{free} (chromatic, no constraints);
    \val{modal} (a mode held for several phrases, e.g.\ D Dorian);
    \val{progression} (cycling ii--V--I or blues);
    \val{pedal} (fixed bass note with shifting upper harmony).
  \item[scale \val{ARC / BASS / BLEND}]
    Where the sax's scale came from.
    \val{ARC} = the arc's harmonic plan.
    \val{BASS} = the bassist implied 4+ distinct pitch classes and the sax
    followed them directly.
    \val{BLEND} = arc scale broadened to include bass pitch classes.
  \item[\texttt{->} + contour]
    The melodic direction the sax aimed for: \val{ascending},
    \val{descending}, or \val{neutral}.
\end{description}

\vspace{6pt}
\noindent\textbf{Bottom row:}

\vspace{2pt}
\begin{description}
  \item[lead \val{bass} / \val{sax}]
    Who is leading the conversation.
    \val{bass} = sax is responding and following;
    \val{sax} = sax is leading and initiating.
  \item[chord \val{name}]
    The chord the LSTM was conditioned on for this phrase ---
    e.g.\ \val{Dm}, \val{G7}, \val{Cmaj}, \val{NC} (no chord).
    Minor chords represent the whole minor family (m, m7, m9, etc.);
    similarly for other qualities.
  \item[vel \val{N}]
    Base MIDI velocity for the phrase (mirrors the bassist's dynamic level).
  \item[n \val{N}]
    Number of notes in the generated phrase.
  \item[note names (cyan)]
    The pitch classes of the generated phrase in order.
\end{description}

  \item[contour] Melodic direction aimed for across recent phrases.
\end{description}
